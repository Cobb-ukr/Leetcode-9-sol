class Solution {
    public boolean isPalindrome(int x) {
        int a=x;
        int rev=0;
        while(a!=0)
        {
            int num= a%10;
            rev= (rev*10)+num;
            a=a/10;
        }
        
        if(rev==x && x>=0)
        {
            return true;
        }
        else
            return false;
        }
    }
